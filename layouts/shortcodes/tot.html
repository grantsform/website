{{/* tot shortcode — a single post within a thread */}} {{- $dateStr := .Get "date" -}} {{- $date := time $dateStr -}} {{- $raw := .Inner -}} {{- $id := .Get "id" -}} {{- /* If no explicit id, allow a standalone line like `#intro` in the tot body to act as an id marker. */ -}} {{- if not $id }} {{- $standalones := findRE `(?m)^[ \t]*#[A-Za-z][A-Za-z0-9_-]*[ \t]*$` $raw -}} {{- if gt (len $standalones) 0 }} {{- $line := index $standalones 0 -}} {{- $line = strings.TrimSpace $line -}} {{- $line = strings.TrimPrefix $line "#" -}} {{- $id = $line -}} {{- $raw = replaceRE `(?m)^[ \t]*#[A-Za-z][A-Za-z0-9_-]*[ \t]*$` "" $raw -}} {{- end -}} {{- end -}} {{- $refsRaw := .Get "refs" | default "" -}} {{- $refs := slice -}} {{- if ne $refsRaw "" }} {{- $refs = split (replaceRE "\\s+" "" $refsRaw) "," -}} {{- end -}} {{- /* render inner markdown to HTML then linkify hashtags */ -}} {{- $html := $raw | markdownify -}} {{- $tags := findRE "#[A-Za-z][A-Za-z0-9_]+" $raw -}} {{- range $tags }} {{- $name := strings.TrimPrefix "#" . -}} {{- $link := (printf "/tots/?tag=%s" ($name | urlize)) -}} {{- $anchor := (printf "#%s" $link $name) -}} {{- $html = replace $html . $anchor -}} {{- end }}
{{ $html | safeHTML }}
